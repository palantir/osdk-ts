{/*
  One-pager docs. The title + description come from the YAML frontmatter of
  ObjectTableOverview.md; this MDX renders a single heading, the live demo, and
  the Markdown body below it. The floating "on this page" table of contents is
  provided by addon-docs (docs.toc in .storybook/preview.tsx), which scans the
  h2/h3 headings rendered from the Markdown body — so there is no hand-authored
  contents list here. The original ObjectTable.md is the full reference and is
  intentionally left untouched.
*/}

import { Canvas, Markdown, Meta } from "@storybook/addon-docs/blocks";
import objectTableDocs from "@docs/ObjectTableOverview.md?raw";
import * as ObjectTableStories from "./ObjectTable.stories";

<Meta title="Components/ObjectTable/Docs" tags={["beta"]} />

export const doc = (() => {
  // Split YAML frontmatter (--- title / description ---) from the Markdown body
  // so this MDX owns the single rendered <h1> + description.
  const match = objectTableDocs.match(/^---\n([\s\S]*?)\n---\n/);
  const meta = {};
  let body = objectTableDocs;
  if (match) {
    body = objectTableDocs.slice(match[0].length);
    for (const line of match[1].split("\n")) {
      const idx = line.indexOf(":");
      if (idx > 0) meta[line.slice(0, idx).trim()] = line.slice(idx + 1).trim();
    }
  }
  return {
    title: meta.title ?? "",
    description: meta.description ?? "",
    body,
  };
})();

<h1>{doc.title}</h1>

<p>{doc.description}</p>

<Canvas of={ObjectTableStories.Default} />

<Markdown>{doc.body}</Markdown>
