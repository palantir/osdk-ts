export const Example = ({ raw, story }) => {
  // Split the chunk at its first fenced code block: prose (### title +
  // description) renders above the live demo; the code block and the
  // "Open the … story →" link (authored in the chunk .md) render below it.
  const fence = raw.indexOf("\n```");
  const intro = fence === -1 ? raw : raw.slice(0, fence);
  const code = fence === -1 ? "" : raw.slice(fence + 1);
  return (
    <>
      <Markdown>{intro}</Markdown>
      {story ? <Canvas of={story} sourceState="none" /> : null}
      {code ? <Markdown>{code}</Markdown> : null}
    </>
  );
};
